\section{傅里叶级数}

\begin{equation}
    s(t_1,t_2)=a_{00}+\sum_{n=1}^{\infty}\sum_{m=1}^{\infty} \left( a_{mn}\cos(m \omega_1 t_1) \right)
\end{equation}
